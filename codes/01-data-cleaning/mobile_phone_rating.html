#create label
# Function to determine buying intent based on rank
def determine_buying_intent(Rank):
    if Rank <= 50:
        return 'very_willing'
    elif 51 <= Rank <= 100:
        return 'moderately_willing'
    else:
        return 'not_willing'
df['buying_intent'] = df['Rank'].apply(determine_buying_intent)
